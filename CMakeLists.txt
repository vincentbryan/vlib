cmake_minimum_required(VERSION 3.9)
project(vlib)

find_package(Threads REQUIRED)

set(VLIB_ROOT ${PROJECT_SOURCE_DIR})

        ${VLIB_ROOT}/include
		${VLIB_ROOT}/3rd/Eigen/include
)

target_link_libraries(vlib
		Threads::Threads
)
